25pub struct HeifContext<'a> {

27    source: Source<'a>,

30impl HeifContext<'static> {
31    /// Create a new empty context.
32    pub fn new() -> Result<HeifContext<'static>> {
33        let ctx = unsafe { lh::heif_context_alloc() };
34        if ctx.is_null() {
35            Err(HeifError {
36                code: HeifErrorCode::ContextCreateFailed,
37                sub_code: HeifErrorSubCode::Unspecified,
38                message: String::from(""),
39            })
40        } else {
41            Ok(HeifContext {
42                inner: ctx,
43                source: Source::None,
44            })
45        }
46    }
47
48    /// Create a new context from file.
49    pub fn read_from_file(name: &str) -> Result<HeifContext<'static>> {
50        let mut context = HeifContext::new()?;
51        context.source = Source::File;
52        let c_name = ffi::CString::new(name).unwrap();
53        let err =
54            unsafe { lh::heif_context_read_from_file(context.inner, c_name.as_ptr(), ptr::null()) };
55        HeifError::from_heif_error(err)?;
56        Ok(context)
57    }
58
59    /// Create a new context from reader.
60    pub fn read_from_reader(reader: Box<dyn Reader>) -> Result<HeifContext<'static>> {
61        let mut context = HeifContext::new()?;
62        let mut reader_box = Box::new(reader);
63        let user_data = reader_box.as_mut() as *mut _ as *mut c_void;
64        let err = unsafe {
65            lh::heif_context_read_from_reader(context.inner, &HEIF_READER, user_data, ptr::null())
66        };
67        HeifError::from_heif_error(err)?;
68        context.source = Source::Reader(reader_box);
69        Ok(context)
70    }
71
72    /// # Safety
73    ///
74    /// The given pointer must be valid.  
75    pub(crate) unsafe fn from_ptr(ctx: *mut lh::heif_context) -> HeifContext<'static> {
76        HeifContext {
77            inner: ctx,
78            source: Source::None,
79        }
80    }
81}

84    /// Create a new context from bytes.
85    ///
86    /// The provided memory buffer is not copied.
87    /// That means, you will have to keep the memory buffer alive as
88    /// long as you use the context.
89    pub fn read_from_bytes(bytes: &[u8]) -> Result<HeifContext> {
90        let mut context = HeifContext::new()?;
91        context.source = Source::Memory(bytes);
92        let err = unsafe {
93            lh::heif_context_read_from_memory_without_copy(
94                context.inner,
95                bytes.as_ptr() as _,
96                bytes.len(),
97                ptr::null(),
98            )
99        };
100        HeifError::from_heif_error(err)?;
101        Ok(context)
102    }

104    unsafe extern "C" fn vector_writer(
105        _ctx: *mut lh::heif_context,
106        data: *const c_void,
107        size: usize,
108        user_data: *mut c_void,
109    ) -> lh::heif_error {
110        let vec: &mut Vec<u8> = &mut *(user_data as *mut Vec<u8>);
111        vec.reserve(size);
112        ptr::copy_nonoverlapping::<u8>(data as _, vec.as_mut_ptr(), size);
113        vec.set_len(size);
114
115        lh::heif_error {
116            code: lh::heif_error_code_heif_error_Ok,
117            subcode: lh::heif_suberror_code_heif_suberror_Unspecified,
118            message: b"\0".as_ptr() as _,
119        }
120    }

122    pub fn write_to_bytes(&self) -> Result<Vec<u8>> {
123        let mut res = Vec::<u8>::new();
124        let pointer_to_res = &mut res as *mut _ as *mut c_void;
125
126        let mut writer = lh::heif_writer {
127            writer_api_version: 1,
128            write: Some(Self::vector_writer),
129        };
130
131        let err = unsafe { lh::heif_context_write(self.inner, &mut writer, pointer_to_res) };
132        HeifError::from_heif_error(err)?;
133        Ok(res)
134    }

174    pub fn top_level_image_handles(&self) -> Vec<ImageHandle> {
175        let max_count = self.number_of_top_level_images();
176        let mut item_ids = Vec::with_capacity(max_count);
177        unsafe {
178            let count = lh::heif_context_get_list_of_top_level_image_IDs(
179                self.inner,
180                item_ids.as_mut_ptr(),
181                max_count as _,
182            ) as usize;
183            item_ids.set_len(count);
184        }
185        let mut handles = Vec::with_capacity(item_ids.len());
186        for item_id in item_ids {
187            if let Ok(handle) = self.image_handle(item_id) {
188                handles.push(handle);
189            }
190        }
191        handles
192    }

194    /// Compress the input image.
195    /// The first image added to the context is also automatically set as the primary image, but
196    /// you can change the primary image later with [`HeifContext::set_primary_image`] method.
197    pub fn encode_image(
198        &mut self,
199        image: &Image,
200        encoder: &mut Encoder,
201        encoding_options: Option<EncodingOptions>,
202    ) -> Result<ImageHandle> {
203        let mut handle: *mut lh::heif_image_handle = ptr::null_mut();
204        unsafe {
205            let err = lh::heif_context_encode_image(
206                self.inner,
207                image.inner,
208                encoder.inner,
209                get_encoding_options_ptr(&encoding_options),
210                &mut handle,
211            );
212            HeifError::from_heif_error(err)?;
213        }
214        Ok(ImageHandle::new(handle))
215    }

217    /// Encode the `image` as a scaled down thumbnail image.
218    /// The image is scaled down to fit into a square area of width `bbox_size`.
219    /// If the input image is already so small that it fits into this bounding
220    /// box, no thumbnail image is encoded and `Ok(None)` is returned.
221    /// No error is returned in this case.
222    ///
223    /// The encoded thumbnail is automatically assigned to the
224    /// `master_image_handle`. Hence, you do not have to call
225    /// [`HeifContext::assign_thumbnail()`] method.
226    pub fn encode_thumbnail(
227        &mut self,
228        image: &Image,
229        master_image_handle: &ImageHandle,
230        bbox_size: u32,
231        encoder: &mut Encoder,
232        encoding_options: Option<EncodingOptions>,
233    ) -> Result<Option<ImageHandle>> {
234        let mut handle: *mut lh::heif_image_handle = ptr::null_mut();
235        unsafe {
236            let err = lh::heif_context_encode_thumbnail(
237                self.inner,
238                image.inner,
239                master_image_handle.inner,
240                encoder.inner,
241                get_encoding_options_ptr(&encoding_options),
242                bbox_size.min(i32::MAX as _) as _,
243                &mut handle,
244            );
245            HeifError::from_heif_error(err)?;
246        }
247        Ok(Some(ImageHandle::new(handle)))
248    }

250    /// Encodes an array of images into a grid.
251    ///
252    /// # Arguments
253    ///
254    /// * `tiles` - User allocated array of images that will form the grid.
255    /// * `rows` - The number of rows in the grid. The number of columns will
256    ///   be calculated from the size of `tiles`.
257    /// * `encoder` - Defines the encoder to use.
258    ///   See [LibHeif::encoder_for_format()](crate::LibHeif::encoder_for_format).
259    /// * `encoding_options` - Optional, may be None.
260    ///
261    /// Returns an error if `tiles` slice is empty.
262    pub fn encode_grid(
263        &mut self,
264        tiles: &[Image],
265        rows: NonZeroU16,
266        encoder: &mut Encoder,
267        encoding_options: Option<EncodingOptions>,
268    ) -> Result<Option<ImageHandle>> {
269        let mut handle: *mut lh::heif_image_handle = ptr::null_mut();
270        let mut tiles_inners: Vec<*mut lh::heif_image> =
271            tiles.iter().map(|img| img.inner).collect();
272        let rows = rows.get();
273        let columns = (tiles_inners.len() as u32 / rows as u32).min(u16::MAX as _) as u16;
274        unsafe {
275            let err = lh::heif_context_encode_grid(
276                self.inner,
277                tiles_inners.as_mut_ptr(),
278                rows,
279                columns,
280                encoder.inner,
281                get_encoding_options_ptr(&encoding_options),
282                &mut handle,
283            );
284            HeifError::from_heif_error(err)?;
285        }
286        Ok(Some(ImageHandle::new(handle)))
287    }

312    /// Add generic, proprietary metadata to an image. You have to specify
313    /// an `item_type` that will identify your metadata. `content_type` can be
314    /// an additional type.
315    ///
316    /// For example, this function can be used to add IPTC metadata
317    /// (IIM stream, not XMP) to an image. Although not standard, we propose
318    /// to store IPTC data with `item_type=b"iptc"` and `content_type=None`.
319    pub fn add_generic_metadata<T>(
320        &mut self,
321        image_handle: &ImageHandle,
322        data: &[u8],
323        item_type: T,
324        content_type: Option<&str>,
325    ) -> Result<()>
326    where
327        T: Into<FourCC>,
328    {
329        let c_item_type = str_to_cstring(&item_type.into().to_string(), "item_type")?;
330        let c_content_type = match content_type {
331            Some(s) => Some(str_to_cstring(s, "content_type")?),
332            None => None,
333        };
334        let c_content_type_ptr = c_content_type.map(|s| s.as_ptr()).unwrap_or(ptr::null());
335        let error = unsafe {
336            lh::heif_context_add_generic_metadata(
337                self.inner,
338                image_handle.inner,
339                data.as_ptr() as _,
340                data.len() as _,
341                c_item_type.as_ptr(),
342                c_content_type_ptr,
343            )
344        };
345        HeifError::from_heif_error(error)
346    }

374    /// If the maximum threads number is set to 0, the image tiles are
375    /// decoded in the main thread. This is different from setting it to 1,
376    /// which will generate a single background thread to decode the tiles.
377    ///
378    /// Note that this setting only affects `libheif` itself. The codecs itself
379    /// may still use multi-threaded decoding. You can use it, for example,
380    /// in cases where you are decoding several images in parallel anyway you
381    /// thus want to minimize parallelism in each decoder.
382    pub fn set_max_decoding_threads(&mut self, max_threads: u32) {
383        let max_threads = max_threads.min(libc::c_int::MAX as u32) as libc::c_int;
384        unsafe { lh::heif_context_set_max_decoding_threads(self.inner, max_threads) };
385    }
