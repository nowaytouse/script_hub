22#[derive(Debug, Copy, Clone, PartialEq, Eq, Hash)]
23pub enum ColorSpace {
24    Undefined,
25    YCbCr(Chroma),
26    Rgb(RgbChroma),
27    Monochrome,
28}
29
30impl ColorSpace {
31    pub(crate) fn from_libheif(
32        color_space: lh::heif_colorspace,
33        chroma: lh::heif_chroma,
34    ) -> Option<Self> {
35        match color_space {
36            lh::heif_colorspace_heif_colorspace_undefined => Some(ColorSpace::Undefined),
37            lh::heif_colorspace_heif_colorspace_monochrome => Some(ColorSpace::Monochrome),
38            lh::heif_colorspace_heif_colorspace_YCbCr => match chroma {
39                lh::heif_chroma_heif_chroma_420 => Some(ColorSpace::YCbCr(Chroma::C420)),
40                lh::heif_chroma_heif_chroma_422 => Some(ColorSpace::YCbCr(Chroma::C422)),
41                lh::heif_chroma_heif_chroma_444 => Some(ColorSpace::YCbCr(Chroma::C444)),
42                _ => None,
43            },
44            lh::heif_colorspace_heif_colorspace_RGB => match chroma {
45                lh::heif_chroma_heif_chroma_444 => Some(ColorSpace::Rgb(RgbChroma::C444)),
46                lh::heif_chroma_heif_chroma_interleaved_RGB => {
47                    Some(ColorSpace::Rgb(RgbChroma::Rgb))
48                }
49                lh::heif_chroma_heif_chroma_interleaved_RGBA => {
50                    Some(ColorSpace::Rgb(RgbChroma::Rgba))
51                }
52                lh::heif_chroma_heif_chroma_interleaved_RRGGBB_BE => {
53                    Some(ColorSpace::Rgb(RgbChroma::HdrRgbBe))
54                }
55                lh::heif_chroma_heif_chroma_interleaved_RRGGBB_LE => {
56                    Some(ColorSpace::Rgb(RgbChroma::HdrRgbLe))
57                }
58                lh::heif_chroma_heif_chroma_interleaved_RRGGBBAA_BE => {
59                    Some(ColorSpace::Rgb(RgbChroma::HdrRgbaBe))
60                }
61                lh::heif_chroma_heif_chroma_interleaved_RRGGBBAA_LE => {
62                    Some(ColorSpace::Rgb(RgbChroma::HdrRgbaLe))
63                }
64                _ => None,
65            },
66            _ => None,
67        }
68    }
69
70    pub(crate) fn heif_color_space(self) -> lh::heif_colorspace {
71        match self {
72            ColorSpace::YCbCr(_) => lh::heif_colorspace_heif_colorspace_YCbCr,
73            ColorSpace::Rgb(_) => lh::heif_colorspace_heif_colorspace_RGB,
74            ColorSpace::Monochrome => lh::heif_colorspace_heif_colorspace_monochrome,
75            ColorSpace::Undefined => lh::heif_colorspace_heif_colorspace_undefined,
76        }
77    }
78
79    pub(crate) fn heif_chroma(self) -> lh::heif_chroma {
80        match self {
81            ColorSpace::YCbCr(chroma) => match chroma {
82                Chroma::C420 => lh::heif_chroma_heif_chroma_420,
83                Chroma::C422 => lh::heif_chroma_heif_chroma_422,
84                Chroma::C444 => lh::heif_chroma_heif_chroma_444,
85            },
86            ColorSpace::Rgb(chroma) => match chroma {
87                RgbChroma::C444 => lh::heif_chroma_heif_chroma_444,
88                RgbChroma::Rgb => lh::heif_chroma_heif_chroma_interleaved_RGB,
89                RgbChroma::Rgba => lh::heif_chroma_heif_chroma_interleaved_RGBA,
90                RgbChroma::HdrRgbBe => lh::heif_chroma_heif_chroma_interleaved_RRGGBB_BE,
91                RgbChroma::HdrRgbLe => lh::heif_chroma_heif_chroma_interleaved_RRGGBB_LE,
92                RgbChroma::HdrRgbaBe => lh::heif_chroma_heif_chroma_interleaved_RRGGBBAA_BE,
93                RgbChroma::HdrRgbaLe => lh::heif_chroma_heif_chroma_interleaved_RRGGBBAA_LE,
94            },
95            ColorSpace::Undefined => lh::heif_chroma_heif_chroma_undefined,
96            ColorSpace::Monochrome => lh::heif_chroma_heif_chroma_monochrome,
97        }
98    }
99}
